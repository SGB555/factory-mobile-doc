# popup-radio

这是由nf-select组件和vux框架的popup组件 再次封装而成的组件，多用于弹窗选项

<popup-radio :options='supplierList' title="物料供应商" selectKey='id' label='name' v-model="createdParams.supplierId" @input='getPurchaseList' required>
    <div slot="right" class="selectedVal grey">
        {{supplierName}}
        <i class="icon-arrow-right action-icon"></i>
    </div>
</popup-radio>

props: {
    options: {
        type: Array || Object,
        default: () => [],
        required: true
    },
    // 带选项
    label: {
        type: String,
        required: true
    },
    // 可选项的显示文字
    selectKey: {
        type: String,
        required: true
    },
    // 可选项数据结构中的key值。例如：[{val:'123'}],就传入val
    popHead: {
        type: String
    },
    // 弹窗显示的标题，如图中的’请选择相关人员‘
    title: {
        type: String
    },
    // 需要选择的相关内容 如图中的’物料供应商‘
    multiple: {
        type: Boolean,
        default: false
    },
    // 是否多选
    value: {},
    required: {
        type: Boolean,
        default: false
    }
    // 是否必填
},
