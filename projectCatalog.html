# 项目文件结构

本项目是通过vue-cli生成的，同时含有cordova构建项目需要的文件

factory-mobile
├── 分支热更新配置说明.txt
├── node_modules
├── package.json
├── .gitignore
├── cordova-hcp.json // cordova项目热更新配置文件
├── config.xml // cordova项目热更新配置文件
├── .gitattributes
├── .editorconfig
├── .babelrc
├── .babelrc
├── www/ // cordova构建项目打包出的文件夹
├── static/ // cordova构建项目打包出的文件夹
├── src
    ├── api // 网络请求(在写新请求的时候根据功能模块可以分出新文件，对于类似通用的请求可以先查找是否已经写入避免重复声明)
        ├── index.js // 项目通用请求 也是一份过于冗余的文件
        ├── ligation.js // 扎号相关请求
        ├── material.js // 物料相关请求
        └── wym.js // 唯一码相关请求
    ├── assets
        ├── icon // 项目中使用的小icon均放至此文件夹
        ├── img // 较大的图片
        ├── style
            ├── common.less // 通用样式
            ├── el-table.less // 用于修改引入的element-ui的table组件的样式文件
            ├── icon.less // 项目中用到的图标都是在这个文件引入放置在../icon 的小图标，具体可参照源代码
            ├── reset.less 
            ├── resetVux.less // 重置vux的样式文件 
            ├── var.less // 声明的样式通用变量 
            └── warning.less
        └── svg/ 
    ├── components 
        ├── common // 通用组件
        ├── home // home.vue通过头部和导航栏
        ├── ligation // 扎号通用组件
        ├── material-buy // 物料采购相关组件
        ├── material-delivery // 送料相关组件
        ├── material-file // 物料档案相关组件
        ├── message // 消息页相关组件
        ├── mission // 工单相关组件
        ├── node // 工单节点相关组件
        ├── report // 上报页相关组件
        ├── report-record // 上报记录相关组件
    ├── config // 各种配置文件
        ├── axios.js // axios配置文件
        ├── busEvent.js // umm.. 这之前同事留下的坑 既然我们都使用了vuex了就不需要eventbus来管理状态了，统一使用vuex吧
        ├── index.js // 上传照片配置文件
        └── rongyun.js // 融云配置文件
    ├── directives/ 自定义指令
    ├── router
        ├── ligation /
        ├── material /
        ├── material-arrive /
        ├── material-buy /
        ├── material-delivery /
        ├── material-file /
        ├── material-refund /
        ├── mission /
        ├── node /
        ├── setting /
        └── index.js
    ├── utils // 工具函数
        └── aboutTable.js // 用于构建SKU表格所需的数据结构
        ├── utils // 工具函数
        └── aboutTable.js // 用于构建SKU表格所需的数据结构
    ├── views / 各页面文件夹
    ├── vuex
    ├── app.vue
    └── main.js
├── res / app 图片以及启动页
├── plugins / cordova 插件
├── platforms / cordova ios和安卓打包所需文件
├── node_modules
├── config // 4个环境的请求BASE_URL
└── build // webpack相关文件
